compra

idcompra PK int(11)
numerodocumento varchar(25)

fecha date
observacion varchar(255)
idmoneda int(11) FK >- moneda.idmoneda
iddeposito int(11) FK >- deposito.iddeposito
idtipomovimiento int(11) FK >- tipo_movimiento.idtipomovimiento
idproveedor int(11) FK >- proveedor.idproveedor
idusuario int(11) FK >- usuario.idusuario
totalneto double
totaliva double
idcuenta int(11) FK >- cuenta.idcuenta
idempresa int(11) FK >- empresa.idempresa
idsucursal int(11) FK >- sucursal.idsucursal
